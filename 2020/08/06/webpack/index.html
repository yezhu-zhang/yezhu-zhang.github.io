1.webpack

webpack是基于node.js的一个自动任务的运行器,打包的工具
webpack是为react而量身打造的一款模块化开发工具

2.webpack的安装与版本

安装

cnpm i webpack -g 
cnpm i webpack webpack-cil -S

版本查看

webpack -v

4以上的版本免配但是可配
4版本以下的必须配置

2.1 0配置

免配置
在一个目录src文件夹下有一个index.js文件,在index.js里编写业务逻辑,命令行执行

webpack

自动打包出dist/main.js

设置运行开发模式

webpack --mode development

2.2 改变默认入口

将入口文件改为src下的test.js

webpack src/test.js --mode production

2.3 改变默认的出口

将出口文件改为dist下的bundle.js

webpack src/test.js --output dist/bundle.js --mode production

2.4 多入口

将src下的one.js和two.js都变为入口文件

webpack src/one.js src/two.js --mode development

3.webpack配置

入口 – entry
出口 – output
插件 — plugins (压缩,页面模板,去除引号等)
转换器 — loader (scss/less.stylus/scc/js)
服务器 – devServer (host/port/代理/默认打开浏览器)
别名,缺省或追名 – resolve

根目录下创建文件 webpack.config.js

module.exports = {}

3.1 入口和出口配置

module.exports = {
  mode: 'development',
  entry: {
    app: './src/main.js'// 入口文件,多入口可以写多个
  },
  output: { // 必须写成对象形式
    path: __dirname+'/dist',
    filename: '[name].[hash:8].js'//防止缓存,添加八位随机数
  }
 }

3.2 页面模板

打包时自动吧public下的index.html文件打包到dist目录,并且自动引入导出的js文件
插件
每次打包前先清空

cnpm i clean-webpack-plugin -S

打包页面模板

cnpm i html-webpack-plugin-S
var HtmlWebpackPlugin = require('html-webpack-plugin') // 导入打包页面模板插件
var { CleanWebpackPlugin } = require('clean-webpack-plugin') // 导入清空插件
module.exports = {
  mode: 'development',
  entry: {
    app: './src/main.js'
  },
  output: {
    path: __dirname+'/dist',
    filename: '[name].[hash:8].js'
  },
  plugins: [
    new CleanWebpackPlugin(), // 打包时替换上一次打包的文件
    new HtmlWebpackPlugin({
      template: './public/index.html', // 自动把public.index,html文件复制到dist/index.html,并且自动引入了js文件,默认在body的结尾引入
      minify: 
        removeAttributeQuotes: true, // 移除属性的符号
        removeEmptyAttributes: true, // 移除空的属性
        removeComments: true, // 移除注释
        collapseWhitespace: true, // 折叠空白部分
        // removeEmptyElement:true // 移除空元素 --一般不设定
        removeScriptTypeAttributesL: true // 引入导入的script 的type的类型
      }
    })
  ]
}

4.转换器

loader让webpack能够去处理那些非JavaScript文件(webpack自身只理解JavaScript).loader可以将所有类型的文件转换为webpack能够处理的有效模块,然后就可以利用webpack的打包能力对它们进行处理

4.1 处理css类型文件

cnpm i css-loader style-loader -S
const { CleanWebpackPlugin } = require('clean-webpack-plugin')
const VueLoaderPlugin = require('vue-loader/lib/plugin')
module.exports = {
  mode: 'development',
  entry: {
      ...
  },
  output: {
      ...
  },
  plugins: [
      ...
  ],
  // 处理非js类型的文件转换为在 项目 可以运行的代码
  module: {
    rules: [
      {
        test: /\.css$/, // 正则表达式表示以css结尾的文件
        // loader: 'style-loader!css-loader'
        use: [
          // { loader: 'style-loader' },
          { loader: 'style-loader'},
          {
            loader: 'css-loader',
            options: {
              modules: true
            }
          }
        ]
      },

4.2 处理scss文件

cnpm i sass node-sass sass-loader -S
  {

      {

        }

4.4 处理js类型的高级语法

cnpm i @babel/core babel-loader -S
cnpm i @babel/preset-env @babel/preset-react -S
  {
    test: /\.(js|jsx)$/,
    // loader: 'babel-loader' // 根目录下创建  .babelrc 文件 { presets: []} 
    use: [{
      loader: 'babel-loader',
      options: {
        // 前者为 js高级语法服务，后者为react服务
        presets: ['@babel/preset-env', '@babel/preset-react']
      }
    }]
  }

          outputPath: 'img'

      }
    ]
  }

5.开发者服务器

cnpm i webpack-dev-server -g
cnpm i webpack-dev-server -S
webpack-dev-server

配置运行命令
打开根目录下的packgae.json

  "scripts": {
    "dev": "webpack-dev-server",
    "build": "webpack"
  },

开发命令

cnpm run dev

打包命令

cnpm run build

5.1 配置开发者服务器

 devServer: {
    host: '0.0.0.0', // 局域网内访问别人的项目
    // open: true // 自动打开默认浏览器运行项目
    proxy: { // 配置代理的服务器 --- 解决跨域问题
    } 
  }

5.2运行开到源码和@符号

devtool: 'source-map', // 方便查看源代码 --  代码调试
  // @
  resolve: {
    extensions: ['.js', '.json', '.stylus', '.vue'], // 引入时可以省略后缀名
    alias: { // @ 代表src目录
      '@': __dirname + '/src'
    }
  }
}
