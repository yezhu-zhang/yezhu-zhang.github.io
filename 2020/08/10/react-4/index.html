配置基本路由 App.js

import React from 'react';
// BrowserRouter   /        history
// HashRouter      /#/      hash
import {
  BrowserRouter as Router,
  Switch,
  Route,
  Redirect,
  NavLink
} from 'react-router-dom'

import Home from './views/home'
import Kind from './views/kind'
import Cart from './views/cart'
import User from './views/user'
// exact 精准匹配路由
function App() {
  return (
    <Router>
      <div>
        <NavLink to="/home">首页</NavLink>
        <NavLink to="/kind">分类</NavLink>
        <NavLink to="/cart">购物车</NavLink>
        <NavLink to="/user">我的</NavLink>
      </div>
      <hr />
      <Switch>
        <Route path="/home" component = { Home } />
        <Route path="/kind" >
          <Kind />
        </Route>
        <Route path="/cart" render = { () => {
          return (
            <Cart />
            )
          }} />
        <Route path="/user" component = { User } />
        <Redirect path="/" exact to="/home" />
      </Switch>
    </Router>
  );
}

export default App;

router/RouterView.jsx

import React from 'react'
import { HashRouter as Router, Switch, Route, Redirect } from 'react-router-dom'
import routes from './index'
class RouterView extends React.Component {
  render () {
    return (
      <Router>
        <Switch>
          {
            // routes.map(item => {
            //   return (
            //     <Route key= { item.path } path={ item.path } component = { item.components.default } />
            //   )
            // })
            routes.map(item => {
              return (
                <Route key = { item.path } path={ item.path } >
                  {
                    this.props.name === 'default' || !this.props.name ? item.components.default : null
                  }
                  {
                    this.props.name === 'footer' ? item.components.footer : null
                  }
                </Route>
              )
            })
          }
          <Redirect path="/" exact to="/home" />
        </Switch>
      </Router>
    )
  }
}

export default RouterView

import React from 'react'
import { HashRouter as Router, Switch, Route, Redirect } from 'react-router-dom'
import routes from './index'
class RouterView extends React.Component {
  render () {
    return (
      <Router>
        <Switch>
          {
            // routes.map(item => {
            //   return (
            //     <Route key= { item.path } path={ item.path } component = { item.components.default } />
            //   )
            // })
            routes.map(item => {
              if (item.redirect) {
                return <Redirect  key = { item.path }  path={ item.path } exact to= { item.redirect }/>
              }
              return (
                <Route key = { item.path } path={ item.path } >
                  {
                    this.props.name === 'default' || !this.props.name ? item.components.default : null
                  }
                  {
                    this.props.name === 'footer' ? item.components.footer : null
                  }
                </Route>
              )
            })
          }
          {/* <Redirect path="/" exact to="/home" /> */}
        </Switch>
      </Router>
    )
  }
}

export default RouterView
